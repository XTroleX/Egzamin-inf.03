kw1. SELECT abonamenty.nazwa, abonamenty.cena, abonamenty.opis FROM abonamenty;
kw2. SELECT ROUND(AVG(abonamenty.cena),2) AS Srednio_abonament FROM abonamenty;
kw3. SELECT abonamenty.nazwa, cechy.cecha FROM abonamenty INNER JOIN szczegolyabonamentu ON abonamenty.id = szczegolyabonamentu.Abonamenty_id INNER JOIN cechy ON szczegolyabonamentu.Cechy_id = cechy.id WHERE abonamenty.id = 1;
